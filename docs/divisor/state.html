103    def from_cli_args(
104        cls,
105        prompt: str,
106        width: int,
107        height: int,
108        num_steps: int,
109        guidance: float,
110        seed: int = 0,
111        neg_prompt: str | None = None,
112        current_sample: torch.Tensor | None = None,
113        timesteps: List[float] | None = None,
114        **kwargs,
115    ) -> "DenoisingState":
116        """Create DenoisingState from CLI arguments.
117
118        :param prompt: Text prompt
119        :param width: Image width
120        :param height: Image height
121        :param num_steps: Number of denoising steps
122        :param guidance: Guidance scale
123        :param seed: Random seed
124        :param neg_prompt: Negative prompt
125        :param current_sample: Initial sample tensor (if available, otherwise empty tensor)
126        :param timesteps: List of timesteps (if available, used to determine total_timesteps)
127        :param kwargs: Additional fields to set (e.g., deterministic, vae_shift_offset, etc.)
128        :returns: DenoisingState initialized from CLI args
129        """
130        total_timesteps = len(timesteps) if timesteps else num_steps
131
132        timestep_state = TimestepState(
133            current_timestep=0.0,
134            previous_timestep=None,
135            current_sample=current_sample if current_sample is not None else torch.empty(0),
136            timestep_index=0,
137            total_timesteps=total_timesteps,
138        )
139
140        return cls(
141            timestep=timestep_state,
142            guidance=guidance,
143            width=width,
144            height=height,
145            seed=seed,
146            prompt=prompt,
147            num_steps=num_steps,
148            neg_prompt=neg_prompt,
149            deterministic=bool(torch.get_deterministic_debug_mode()) if "deterministic" not in kwargs else kwargs.pop("deterministic"),
150            **kwargs,
151        )

153    def with_runtime_state(
154        self,
155        current_timestep: float,
156        current_sample: torch.Tensor,
157        timestep_index: int,
158        total_timesteps: int | None = None,
159        previous_timestep: float | None = None,
160    ) -> "DenoisingState":
161        """Create a new DenoisingState with updated runtime state.
162
163        Delegates to TimestepState.with_runtime_state() and replaces the timestep.
164
165        :param current_timestep: Current timestep value
166        :param current_sample: Current sample tensor
167        :param timestep_index: Current timestep index
168        :param total_timesteps: Total number of timesteps (if changed)
169        :param previous_timestep: Previous timestep value
170        :returns: New DenoisingState with updated runtime fields
171        """
172        new_timestep = self.timestep.with_runtime_state(
173            current_timestep=current_timestep,
174            current_sample=current_sample,
175            timestep_index=timestep_index,
176            total_timesteps=total_timesteps,
177            previous_timestep=previous_timestep,
178        )
179        return replace(self, timestep=new_timestep)
