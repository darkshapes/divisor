 24def time_shift(
 25    mu: float,
 26    sigma: float,
 27    tensor_step: torch.Tensor,
 28    steps: int,
 29    compress: float = 1.0,
 30) -> torch.Tensor:
 31    """Adjustable noise schedule. Compress or stretch any schedule to match a dynamic step sequence length.
 32
 33    :param mu: Original schedule parameter.
 34    :param sigma: Original schedule parameter.
 35    :param tensor_step: Tensor of original timesteps in [0,1].
 36    :param steps: Desired number of timesteps.
 37    :param compress: >1 compresses (fewer steps), <1 stretches (more steps).
 38    :returns: Adjusted timestep tensor.
 39    """
 40    # Handle edge case where steps is 1
 41    if steps == 1:
 42        return tensor_step
 43
 44    # map original tensor_step to new index space
 45    new_idx = (tensor_step * (steps - 1) * compress).clamp(0, steps - 1)
 46    # rescale back to [0,1]
 47    t_adj = new_idx / (steps - 1)
 48    # Avoid division by zero
 49    t_adj = torch.clamp(t_adj, min=1e-8, max=1.0 - 1e-8)
 50    return torch.exp(torch.tensor(mu)) / (torch.exp(torch.tensor(mu)) + (1 / t_adj - 1) ** torch.tensor(sigma))

 53def serialize_state_for_chain(state: "DenoisingState", current_seed: int) -> str:
 54    """Serialize DenoisingState for HyperChain storage, excluding current_sample and adding current_seed.
 55
 56    :param state: The DenoisingState to serialize
 57    :param current_seed: The current seed value to include instead of current_sample
 58    :returns: JSON string representation of the state
 59    """
 60    state_dict = asdict(state)
 61    # Remove the tensor (current_sample) from nested timestep as it's not serializable
 62    if "timestep" in state_dict and isinstance(state_dict["timestep"], dict):
 63        state_dict["timestep"].pop("current_sample", None)
 64    # Flatten timestep fields for backward compatibility
 65    if "timestep" in state_dict and isinstance(state_dict["timestep"], dict):
 66        timestep_dict = state_dict.pop("timestep")
 67        state_dict.update(timestep_dict)
 68    # Add the seed instead
 69    state_dict["current_seed"] = current_seed
 70    return json.dumps(state_dict, default=str)

105class ManualTimestepController:
106    """
107    Controller for manually stepping through denoising timesteps.
108
109    Instead of automatically processing all timesteps, this allows
110    the user to increment timesteps one at a time, with the ability
111    to intervene between steps.
112    """
113
114    hyperchain: HyperChain = HyperChain()
115
116    def __init__(
117        self,
118        timesteps: list[float],
119        initial_sample: Any,
120        denoise_step_fn: Callable[[Any, float, float, float], Any],
121        mu: float = 0.0,
122        sigma: float = 1.0,
123        initial_guidance: float = 7.5,
124    ):
125        """Initialize the controller.
126
127        :param timesteps: List of timestep values to process (typically from 1.0 to 0.0)
128        :param initial_sample: The initial noisy sample to start denoising from
129        :param denoise_step_fn: Function that performs one denoising step. Signature: (sample, t_curr, t_prev, guidance) -> new_sample
130        :param mu: Schedule parameter for time_shift (default: 0.0)
131        :param sigma: Schedule parameter for time_shift (default: 1.0)
132        :param initial_guidance: Initial guidance (CFG) value (default: 7.5)
133        :param hyperchain: Optional HyperChain instance for storing state history (default: None)
134        """
135        self.timesteps = timesteps
136        self.original_timesteps = timesteps.copy()
137        self.denoise_step_fn = denoise_step_fn
138        self.current_index = 0
139        self.current_sample = initial_sample
140        self.state_history: list[DenoisingState] = []
141        self.mu = mu
142        self.sigma = sigma
143        self.guidance = initial_guidance
144        self.guidance_history: list[float] = [initial_guidance]
145        self.layer_dropout: Optional[list[int]] = None
146        self.layer_dropout_history: list[Optional[list[int]]] = [None]
147        self.width: Optional[int] = None
148        self.height: Optional[int] = None
149        self.seed: Optional[int] = None
150        self.prompt: Optional[str] = None
151        self.num_steps: Optional[int] = None
152        self.vae_shift_offset: float = 0.0
153        self.vae_scale_offset: float = 0.0
154        self.use_previous_as_mask: bool = False
155        self.variation_seed: Optional[int] = None
156        self.variation_strength: float = 0.0
157        self.deterministic: bool = False
158
159    @property
160    def is_complete(self) -> bool:
161        """Check if all timesteps have been processed."""
162        return self.current_index >= len(self.timesteps) - 1
163
164    @property
165    def current_state(self) -> DenoisingState:
166        """Get the current state of the denoising process."""
167        t_curr = self.timesteps[self.current_index]
168        t_prev = self.timesteps[self.current_index + 1] if self.current_index + 1 < len(self.timesteps) else None
169
170        timestep = TimestepState(
171            current_timestep=t_curr,
172            previous_timestep=t_prev,
173            current_sample=self.current_sample,
174            timestep_index=self.current_index,
175            total_timesteps=len(self.timesteps),
176        )
177
178        return DenoisingState(
179            timestep=timestep,
180            guidance=self.guidance,
181            layer_dropout=self.layer_dropout,
182            width=self.width,
183            height=self.height,
184            seed=self.seed,
185            prompt=self.prompt,
186            num_steps=self.num_steps,
187            vae_shift_offset=self.vae_shift_offset,
188            vae_scale_offset=self.vae_scale_offset,
189            use_previous_as_mask=self.use_previous_as_mask,
190            variation_seed=self.variation_seed,
191            variation_strength=self.variation_strength,
192            deterministic=self.deterministic,
193        )

195    def step(self) -> DenoisingState:
196        """Manually increment to the next timestep and perform one denoising step. Uses the current guidance value.
197
198        :returns: The new state after the step.
199        :raises ValueError: If all timesteps have already been processed.
200        """
201        if self.is_complete:
202            raise ValueError("All timesteps have been processed. Cannot step further.")
203
204        t_curr = self.timesteps[self.current_index]
205        t_prev = self.timesteps[self.current_index + 1]
206
207        # Perform the denoising step with current guidance
208        self.current_sample = self.denoise_step_fn(self.current_sample, t_curr, t_prev, self.guidance)
209
210        # Move to next timestep
211        self.current_index += 1
212
213        # Save state and guidance
214        state = self.current_state
215        self.state_history.append(state)
216        self.guidance_history.append(self.guidance)
217        self.layer_dropout_history.append(self.layer_dropout)
218
219        return state

238        self.layer_dropout_history = [self.layer_dropout]

255    def stretch_compress_schedule(
256        self,
257        compress: float,
258        steps: Optional[int] = None,
259    ) -> list[float]:
260        """Stretch or compress the remaining timesteps in the schedule using time_shift.
261
262        :param compress: >1 compresses (fewer steps), <1 stretches (more steps)
263        :param steps: Desired number of timesteps for the remaining schedule. If None, uses the current number of remaining timesteps.
264        :returns: New list of timesteps for the remaining schedule.
265        """
266        if self.is_complete:
267            return []
268
269        # Get remaining timesteps
270        remaining_timesteps = self.timesteps[self.current_index :]
271
272        if steps is None:
273            steps = len(remaining_timesteps)
274
275        # Convert to tensor and normalize to [0, 1] if needed, assuming timesteps are in [0, 1] range
276        tensor_steps = torch.tensor(remaining_timesteps, dtype=torch.float32)
277        adjusted_timesteps = time_shift(self.mu, self.sigma, tensor_steps, steps, compress)
278        new_timesteps = adjusted_timesteps.tolist()
279
280        # Update the schedule (keep processed timesteps, replace remaining)
281        self.timesteps = self.timesteps[: self.current_index] + new_timesteps
282
283        return new_timesteps

285    def stretch_compress_current_step(
286        self,
287        sub_steps: int,
288        compress: float = 1.0,
289    ) -> list[float]:
290        """Stretch or compress the current step by subdividing it into multiple steps. This allows finer control over a single denoising step.
291
292        :param sub_steps: Number of sub-steps to divide the current step into
293        :param compress: >1 compresses the sub-steps (closer spacing), <1 stretches them (wider spacing). Affects the distribution of sub-steps.
294        :returns: List of new timesteps that replace the current step.
295        """
296        if self.is_complete:
297            raise ValueError("No current step to subdivide.")
298
299        t_curr = self.timesteps[self.current_index]
300        t_prev = self.timesteps[self.current_index + 1] if self.current_index + 1 < len(self.timesteps) else 0.0
301
302        if t_curr == t_prev:
303            # No step to subdivide
304            return []
305
306        # Create evenly spaced sub-timesteps between t_curr and t_prev
307        sub_timesteps = torch.linspace(t_curr, t_prev, sub_steps + 1)
308
309        # Apply compression/stretching using time_shift if needed
310        if compress != 1.0:
311            # Normalize to [0, 1] range where 1.0 = t_curr and 0.0 = t_prev
312            # This maps the interval [t_prev, t_curr] to [0, 1]
313            normalized = (sub_timesteps - t_prev) / (t_curr - t_prev)
314
315            # Apply time_shift to adjust the spacing
316            adjusted_normalized = time_shift(self.mu, self.sigma, normalized, sub_steps + 1, compress)
317
318            # Denormalize back to original range [t_prev, t_curr]
319            sub_timesteps = adjusted_normalized * (t_curr - t_prev) + t_prev
320
321            # Ensure we maintain the endpoints
322            sub_timesteps[0] = t_curr
323            sub_timesteps[-1] = t_prev
324
325        # Convert to list (exclude the first one as it's the current timestep)
326        new_sub_timesteps = sub_timesteps[1:].tolist()
327
328        # Replace current step with sub-steps
329        self.timesteps = self.timesteps[: self.current_index + 1] + new_sub_timesteps + self.timesteps[self.current_index + 1 :]
330
331        return new_sub_timesteps

286    def stretch_compress_current_step(
287        self,
288        sub_steps: int,
289        compress: float = 1.0,
290    ) -> list[float]:
291        """Stretch or compress the current step by subdividing it into multiple steps. This allows finer control over a single denoising step.
292
293        :param sub_steps: Number of sub-steps to divide the current step into
294        :param compress: >1 compresses the sub-steps (closer spacing), <1 stretches them (wider spacing). Affects the distribution of sub-steps.
295        :returns: List of new timesteps that replace the current step.
296        """
297        if self.is_complete:
298            raise ValueError("No current step to subdivide.")
299
300        t_curr = self.timesteps[self.current_index]
301        t_prev = self.timesteps[self.current_index + 1] if self.current_index + 1 < len(self.timesteps) else 0.0
302
303        if t_curr == t_prev:
304            # No step to subdivide
305            return []
306
307        # Create evenly spaced sub-timesteps between t_curr and t_prev
308        sub_timesteps = torch.linspace(t_curr, t_prev, sub_steps + 1)
309
310        # Apply compression/stretching using time_shift if needed
311        if compress != 1.0:
312            # Normalize to [0, 1] range where 1.0 = t_curr and 0.0 = t_prev
313            # This maps the interval [t_prev, t_curr] to [0, 1]
314            normalized = (sub_timesteps - t_prev) / (t_curr - t_prev)
315
316            # Apply time_shift to adjust the spacing
317            adjusted_normalized = time_shift(self.mu, self.sigma, normalized, sub_steps + 1, compress)
318
319            # Denormalize back to original range [t_prev, t_curr]
320            sub_timesteps = adjusted_normalized * (t_curr - t_prev) + t_prev
321
322            # Ensure we maintain the endpoints
323            sub_timesteps[0] = t_curr
324            sub_timesteps[-1] = t_prev
325
326        # Convert to list (exclude the first one as it's the current timestep)
327        new_sub_timesteps = sub_timesteps[1:].tolist()
328
329        # Replace current step with sub-steps
330        self.timesteps = self.timesteps[: self.current_index + 1] + new_sub_timesteps + self.timesteps[self.current_index + 1 :]
331
332        return new_sub_timesteps

292

294        :param compress: >1 compresses the sub-steps (closer spacing), <1 stretches them (wider spacing). Affects the distribution of sub-steps.
295        :returns: List of new timesteps that replace the current step.
296        """
297        if self.is_complete:
298            raise ValueError("No current step to subdivide.")
299
300        t_curr = self.timesteps[self.current_index]
301        t_prev = self.timesteps[self.current_index + 1] if self.current_index + 1 < len(self.timesteps) else 0.0
302
303        if t_curr == t_prev:
304            # No step to subdivide
305            return []
306
307        # Create evenly spaced sub-timesteps between t_curr and t_prev
308        sub_timesteps = torch.linspace(t_curr, t_prev, sub_steps + 1)
309
310        # Apply compression/stretching using time_shift if needed
311        if compress != 1.0:
312            # Normalize to [0, 1] range where 1.0 = t_curr and 0.0 = t_prev
313            # This maps the interval [t_prev, t_curr] to [0, 1]
314            normalized = (sub_timesteps - t_prev) / (t_curr - t_prev)
315
316            # Apply time_shift to adjust the spacing
317            adjusted_normalized = time_shift(self.mu, self.sigma, normalized, sub_steps + 1, compress)
318
319            # Denormalize back to original range [t_prev, t_curr]
320            sub_timesteps = adjusted_normalized * (t_curr - t_prev) + t_prev
321
322            # Ensure we maintain the endpoints
323            sub_timesteps[0] = t_curr
324            sub_timesteps[-1] = t_prev
325
326        # Convert to list (exclude the first one as it's the current timestep)
327        new_sub_timesteps = sub_timesteps[1:].tolist()
328
329        # Replace current step with sub-steps
330        self.timesteps = self.timesteps[: self.current_index + 1] + new_sub_timesteps + self.timesteps[self.current_index + 1 :]
331
332        return new_sub_timesteps
